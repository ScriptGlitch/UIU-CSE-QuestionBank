Question 4: Calculate Playground Perimeter and Fencing Cost
Complete the following program to calculate:
  • The perimeter of a playground
  • The total cost of fencing
Print both the perimeter and the total cost.

Given Code: 

#include <stdio.h>

int main() {
    float playgroundLength, playgroundWidth, fenceCostPerMeter = 15.0;
    float perimeter, totalCost;

    printf("Enter the length of the playground: ");
    scanf("%f", &playgroundLength);
    printf("Enter the width of the playground: ");
    scanf("%f", &playgroundWidth);

    return 0;
}

Corrected Code:
Sample Input Sample Output
Length = 50
Width = 30 		Perimeter = 160.00 meters
Total Cost = $2400.00
Length = 100
Width = 50 		Perimeter = 300.00 meters
Total Cost = $4500.00
Solution 

#include <stdio.h>

int main() {
    float playgroundLength, playgroundWidth, fenceCostPerMeter = 15.0;
    float perimeter, totalCost;

    printf("Enter the length of the playground: ");
    scanf("%f", &playgroundLength);
    printf("Enter the width of the playground: ");
    scanf("%f", &playgroundWidth);

    perimeter = 2 * (playgroundLength + playgroundWidth);
    totalCost = perimeter * fenceCostPerMeter;

    printf("Perimeter = %.2f meters\n", perimeter);
    printf("Total Cost = $%.2f\n", totalCost);

    return 0;
}

Question 5: Internet Data Package Cost Calculator
Write a C program for an Internet Data Package Cost Calculator. The packages are as follows:
  • Basic Package: 50 Taka per GB
  • Standard Package: 100 Taka per GB
  • Premium Package: 150 Taka per GB
The program should:
  1. Ask the user to choose a package by entering 1 for Basic, 2 for Standard, or 3 for Premium.
  2. Ask the user to input the number of GBs they want to use.
  3. Calculate the total cost based on the selected package and data usage.
  4. Print the total cost and the selected package type.

Input 1: 
Select the data package:
1. Basic Package
2. Standard Package
3. Premium Package
Enter your choice: 3
Enter the number of GBs you want to use: 50
Output 1: 
You selected the Premium Package.
The total charge for your package is:
7500.00
Input 2: 
Select the data package:
1. Basic Package
2. Standard Package
3. Premium Package
Enter your choice: 4
Output 2: 
Invalid package type selected.
Solution 

#include <stdio.h>

int main() {
    int packageType;
    float gb, cost;

    printf("Select the data package:\n");
    printf("1. Basic Package\n");
    printf("2. Standard Package\n");
    printf("3. Premium Package\n");
    printf("Enter your choice: ");
    scanf("%d", &packageType);

    if (packageType < 1 || packageType > 3) {
        printf("Invalid package type selected.\n");
        return 0;
    }

    printf("Enter the number of GBs you want to use: ");
    scanf("%f", &gb);

    if (packageType == 1) {
        cost = gb * 50;
        printf("You selected the Basic Package.\n");
    } else if (packageType == 2) {
        cost = gb * 100;
        printf("You selected the Standard Package.\n");
    } else if (packageType == 3) {
        cost = gb * 150;
        printf("You selected the Premium Package.\n");
    }

    printf("The total charge for your package is: %.2f\n", cost);
    return 0;
}

Question 4: Calculate Garden and Road Areas
Complete the following program to calculate:
  • The total area (garden + road) by adding the road width on all sides
  • The area of the road
Given Code: 

#include <stdio.h>

int main() {
    float gardenLength, gardenWidth, roadWidth = 2.0;
    float totalArea, gardenArea, roadArea;

    printf("Enter the length of the garden: ");
    scanf("%f", &gardenLength);
    printf("Enter the width of the garden: ");
    scanf("%f", &gardenWidth);

    gardenArea = gardenLength * gardenWidth;

    return 0;
}

Corrected Code:
Sample Input Sample Output
Garden Length = 10
Garden Width = 5 		Total Area = 112.00 square meters
Garden Area = 50.00 square meters
Road Area = 62.00 square meters
Garden Length = 15
Garden Width = 10 		Total Area = 238.00 square meters
Garden Area = 150.00 square meters
Road Area = 88.00 square meters
Solution 

#include <stdio.h>

int main() {
    float gardenLength, gardenWidth, roadWidth = 2.0;
    float totalArea, gardenArea, roadArea;

    printf("Enter the length of the garden: ");
    scanf("%f", &gardenLength);
    printf("Enter the width of the garden: ");
    scanf("%f", &gardenWidth);

    gardenArea = gardenLength * gardenWidth;
    totalArea = (gardenLength + 2 * roadWidth) * (gardenWidth + 2 * roadWidth);
    roadArea = totalArea - gardenArea;

    printf("Total Area = %.2f square meters\n", totalArea);
    printf("Garden Area = %.2f square meters\n", gardenArea);
    printf("Road Area = %.2f square meters\n", roadArea);

    return 0;
}

Question 5: Transportation Service Charge Calculator
Write a C program for a Transportation Service Charge Calculator. The services are as follows:
  • Car: 500 Taka per kilometer
  • Bus: 300 Taka per kilometer
  • Bike: 100 Taka per kilometer
The program should:
  1. Ask the user to choose a service by entering 1 for Car, 2 for Bus, or 3 for Bike.
  2. Ask the user to input the number of kilometers they want to travel.
  3. Calculate the total charge based on the selected service and distance.
  4. Print the total charge and service type.
Input 1: 
Select the service:
1. Car
2. Bus
3. Bike
Enter your choice: 2
Enter the distance: 2.5
Output 1: 
You selected Bus.
The total charge for your trip is: 750.00
Input 2: 
Select the service:
1. Car
2. Bus
3. Bike
Enter your choice: 5
Enter the distance: 5
Output 2: 
Invalid service type selected.
Solution 

#include <stdio.h>

int main() {
    int serviceType;
    float distance, charge;

    printf("Select the service:\n");
    printf("1. Car\n");
    printf("2. Bus\n");
    printf("3. Bike\n");
    printf("Enter your choice: ");
    scanf("%d", &serviceType);

    if (serviceType < 1 || serviceType > 3) {
        printf("Invalid service type selected.\n");
        return 0;
    }

    printf("Enter the distance (in kilometers): ");
    scanf("%f", &distance);

    if (serviceType == 1) {
        charge = distance * 500;
        printf("You selected Car.\n");
    } else if (serviceType == 2) {
        charge = distance * 300;
        printf("You selected Bus.\n");
    } else if (serviceType == 3) {
        charge = distance * 100;
        printf("You selected Bike.\n");
    }

    printf("The total charge for your trip is: %.2f\n", charge);

    return 0;
}
